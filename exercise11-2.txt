select a.first_name a.last_name, a.slp_instructor_first_name, a.slp_instructor_last_name, b.first_name, b.last_name, b.slp_instructor_first_name, b.slp_instructor_last_name
	from student a join student b
	on a.first_name = b.first_name and a.last_name = b.last_name where a.slp_instructor_first_name < b.slp_instructor_first_name;
